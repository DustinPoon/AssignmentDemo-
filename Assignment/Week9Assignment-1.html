Join the crimes data with holiday data

#join BCHoliday to Crimes set logical holiday
crimPLUSHoli <- crimes %>% 
    left_join(BCHoliday, by = "date")%>%
    mutate(holidaylogi = if_else(is.na(holidayname),FALSE,TRUE),
           DayOfWeek = wday(date, label = TRUE),
           Month = month(date, label = TRUE))
